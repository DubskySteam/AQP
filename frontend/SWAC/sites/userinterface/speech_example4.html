Example 2: Integrate data sources

Search can integrate data sources, those are defined by the options.


<div id="speech_example4" swa="Speech"></div>
        

var speech_example1_options = {
    lang: 'de-DE',
    commands: {
        'de-DE': {
            'rede': {
                speak: 'Hallo'
            },
            'zeige': {
                execute: function() { window.alert('Hallo!');}
            }
        },
        'en-EN': {
            'say': {
                speak: 'Hello'
            },
            'show': {
                execute: function() { window.alert('Hello!');}
            }
        }
    }
};
